  <prepend text="TEXT"/>
  <insert text="TEXT" index="4"/>
  <left length="3"/>
  <right length="3"/>
  <capitalize/>
  <lowercase/>
  <uppercase/>
  <trim/>
  <compress/>
  <strip pattern="TEXT"/>
  <replace pattern="REGEX" replacement="TEXT" scope="all|first"/>
  <substring start="0" length="3"/>
  <substring start="0" end="4"/>
  <substring start="0"/>
  <pad type="left|right" with=" "/>
  
  <transform>
    <boolean format="any|truefalse|yn|yesno|bit"/>
    <number pattern="NUM_PATTERN"/>
    <integer radix="10"/>
    <decimal/>
    <datetime pattern="DT_PATTERN" timezone="TIMEZONE" chronology="CHRONOLOGY"/>
    <extract pattern="REGEX" output="REGEX_BACKREFERENCES"/>
    <join separator=","/>
    <split separator=","/>
    <format pattern="FORMAT_PATTERN"/>
    <!-- any string element from process -->
  </transform>
</process>

REGEX_BACKREFERENCES:
 \0, \1, \2, ...
 \0 indicates the 0th capturing group which is the entire string that was found.
 \1 and greater indicates the 1st capturing group in the specified regex.
